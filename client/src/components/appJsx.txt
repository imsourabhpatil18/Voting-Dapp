import { BrowserRouter as Router, Routes, Route, Link } from "react-router-dom";
import CandidateRegister from "./components/candidateRegister";
import VoterRegister from "./components/voterRegister";
import Winner from "./components/Winner";
import Vote from "./contracts/vote.json";

import Voting from "./components/voting";
import Web3 from "web3";
import { useEffect, useState } from "react";
import "./App.css";
import ElectionCommision from "./components/electionCommision";
import Navbar from "./components/navbar";
import WalletConnect from "./components/walletConect";

function App() {
  const [state, setState] = useState({
    web3: null,
    contract: null,
  });

  useEffect(() => {
    async function init() {
      const contractABI = Vote.abi;

      let provider = new Web3.providers.HttpProvider(
        "https://eth-sepolia.g.alchemy.com/v2/_EemJldI7Q2EbIcwTPXcuKVT_X_UaS5r"
      );

      let web3 = new Web3(provider);

      const Contract = new web3.eth.Contract(
        contractABI,
        "0xfA532Eb4790ff1740891b29f28d1e3fa8CFb8001"
      );

      setState({ web3: web3, contract: Contract });
    }
    init();
  }, []);

  return (
    <>
      <Navbar />
      <WalletConnect state={state} />

      <div className="App">
        <div className="maiNcontainer">
          <Routes>
            <Route path="/voter" element={<VoterRegister state={state} />} />
            <Route
              path="/candidate"
              element={<CandidateRegister state={state} />}
            />
            <Route path="/voting" element={<Voting state={state} />} />
            <Route
              p
              ath="/electioncommision"
              element={<ElectionCommision state={state} />}
            />
            <Route path="/result" element={<Winner state={state} />} />
          </Routes>
        </div>
      </div>
    </>
  );
}

export default App;
